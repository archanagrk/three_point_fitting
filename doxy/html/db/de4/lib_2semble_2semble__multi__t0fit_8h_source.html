1073  void SMT0Fit<T>::reorderStates(void)
1074  {

1076  {
1077  std::cout << __PRETTY_FUNCTION__ << __FILE__ << __LINE__ << " can't reorder w/o init" << std::endl;
1078  exit(1);
1079  }
1080 
1081  reorder.clear();
1082 
1083  const int nops = tp.end()->getN();
1084  const int B = tp.end()->getB();
1085  const int nvecs = max_states;
1086  const int t0_ref = inikeys.t0Props.t0ref;
1087 
1088  if(t0_metrics.empty())
1089  constructMetrics();
1090 
1091 
1092  itpp::Mat<int> zero(nvecs,nvecs);
1093  zero.zeros();
1094  std::vector<itpp::Mat<int> > counts(inikeys.t0Props.t0high - inikeys.t0Props.t0low + 1, zero);
1095 
1096  // use counts to keep a tally for a maximum likelyhood overlap map
1097  // outter vector is t0, inner vector is the ref state, innermost vector is the state on that t0
1098 
1099  // now lets loop over each t,t0 value
1100 
1101  // perhaps this loop should be from t0ref to inikeys.globalProps.tmax?
1102  // then we would have saturation for all the vectors we consider
1103 
1104  for(int t = inikeys.globalProps.tmin; t <= inikeys.globalProps.tmax; ++t)
1105  {
1106  if(t == t0_ref) //no eigenvectors, don't want to contaminate map with junk
1107  continue;
1108 
1109  for(int t0 = inikeys.t0Props.t0low; t0 <= inikeys.t0Props.t0high; ++t0)
1110  {
1111  if(t == t0) //no eigenvectors, don't want to contaminate map with junk
1112  continue;
1113 
1114  //make a map of this t0 at this t to t0_ref at this t
1115  std::map<int,int> mapp = makeRemap(t0_fits[t0_ref]->peekVecs(t),t0_metrics[t0]*t0_fits[t0]->peekVecs(t));
1116  std::map<int,int>::const_iterator it;
1117 
1118  //tally the counts
1119  for(it = mapp.begin(); it != mapp.end(); ++it)
1120  ++counts[t0 - inikeys.t0Props.t0low](it->first,it->second); //increment
1121  }
1122  }
1123 
1124  const int refdim = t0_fits[t0_ref]->getM();
1125 
1126 
1127  /*debug*/ //Put this in a logfile? It isn't actually very useful.
1128 
1129  /*
1130  std::cout << "FOR DEBUGGING" << std::endl;
1131  for(int t0 = inikeys.t0Props.t0low; t0 <= inikeys.t0Props.t0high; ++t0)

1133  << counts[t0 - inikeys.t0Props.t0low] << "\n"<< std::endl;
1134 
1135  */
1136 
1137  // now loop through the maximum likelyhood map to assign states
1138 
1139  // pick out states by the most times they mapped, in the case where the
1140  // operator basis spans the space of the hamiltonian and perfect statistics this
1141  // map would be some permutation of a diagonal matrix
1142 
1143  // since we can't span the space of the hamiltonian we will pick out the most
1144  // likely match according to the maximum number of overlaps
1145 
1146  for(int t0 = inikeys.t0Props.t0low; t0 <= inikeys.t0Props.t0high; ++t0)
1147  {
1148  const int vecdim = t0_fits[t0]->getM();
1149  const int bound = (refdim < vecdim) ? refdim : vecdim;
1150  std::vector<bool> ur(refdim,false),uv(vecdim,false);
1151  int mr,mv,m;
1152  std::map<int,int> rmap_t0;
1153  itpp::Mat<int> dum(counts[t0 - inikeys.t0Props.t0low]);
1154 
1155 
1156  // only pick out the min(refdim,vecdim) states
1157  for(int state = 0; state < bound; ++state)
1158  {
1159  mr = 0;
1160  mv = 0;
1161  m = 0;
1162 
1163  for(int r = 0; r < refdim; ++r)
1164  if(ur[r])
1165  continue; //already used this ref
1166  else
1167  for(int v = 0; v < vecdim; ++v)
1168  {
1169  if(uv[v]) //already used this vec
1170  continue;
1171 
1172  if(dum(r,v) >= m) //pick out the best pair out of the remaining bound - state pairs
1173  {
1174  m = dum(r,v);
1175  mr = r;
1176  mv = v;
1177  }
1178  }
1179 
1180  // cant reuse this pair
1181  ur[mr] = true;
1182  uv[mv] = true;
1183  rmap_t0[mr] = mv;
1184  }
1185 
1186 
1187  /*
1188  CASES
1189  1) refdim = vecdim - do nothing
1190  2) refdim < vecdim - pile anything that didn't map on top with a lowest to lowest convention
1191  3) refdim > vecdim - don't need to do anything, the map takes care of itself, we already
1192  mapped all the states at t0
1193  */
1194 
1195  // fill in the vecdim - refdim states that didn't match in a lowest to lowest convention
1196  if(refdim < vecdim)
1197  {
1198 
1199  // first determine what states did not match (whats available)
1200  ur.clear();
1201  uv.clear();
1202 
1203  ur.resize(vecdim,false);
1204  uv.resize(vecdim,false);
1205 
1206  std::map<int,int>::const_iterator it;
1207 
1208  for(it = rmap_t0.begin(); it != rmap_t0.end(); ++it)
1209  {
1210  ur[it->first] = true;
1211  uv[it->second] = true;
1212  }
1213 
1214  // now load those states into vectors
1215  std::vector<int> rr,vv;
1216 
1217  for(int i = 0; i < vecdim; ++i)
1218  {
1219  if(!!!ur[i])
1220  rr.push_back(i);
1221  if(!!!uv[i])
1222  vv.push_back(i);
1223  }
1224 
1225  // check to make sure we didn't do something stupid.. this can probably be removed now
1226  if(rr.size() != vv.size())
1227  {
1228  std::cout << __PRETTY_FUNCTION__ << __FILE__ << __LINE__ << std::endl;
1229  std::cout << " If you see this something went horribly wrong" << std::endl;
1230  exit(1);
1231  }
1232 
1233  // fill in the map of the unmapped states
1234  for(int i = 0; i < rr.size(); ++i)
1235  rmap_t0[rr[i]] = vv[i];
1236  }
1237 
1238  // the reorder map for this t0
1239  reorder[t0] = rmap_t0;
1240 
1241  }//end t0 loop
1242 
1243  reordered = true;
1244  }
